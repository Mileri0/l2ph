Пример описания формы с таймером.

Обратите внимание на передаваемые в ctreate параметры!!! и на все что отмечено красным.
Скрипт меняющий титул по таймеру, ака бегущая строка.

Unit TitulMain; //название юнита должно совпадать с именем его в проекте.


interface
uses forms, classes, extctrls, sysutils, stdctrls;


const
  Full_Title = '                ЙА креведко, Бугага';//титул полностью - строка любой длины
  Char_Name = 'someone';              //имя перса

type
  TMain = class (tform) //класс родителем которого являеться форма - описывает создание формы.
    Timer : TTimer;  //обьекты распологающиеся на форме.
    Button : TButton;
    Edit: TEdit;                               
   
    procedure ButtonClick(Sender: TObject); // и их события!!!
    procedure TimerTimer(Sender: TObject);
  private
    Pos_i : Integer;  //В секции приват обьявляються переменные доступ к которым
    Start : Boolean;  //есть из методов описаных в самом классе но недоступные извне
                                //это место можно переместить в Public а секцию удалить.
   
  public
    Constructor create(Aowner:Tcomponent); override; //конструктор формы. в нем описываеться создание обьектов распологающихся на форме.
  end;
   
var
  main : TMain; //Экземпляр нашего класса
 
 
Procedure Init;  //функции вызываемые из пх. должны бытть обьявлены тут. т.е. в Interface секции
procedure Free; //реализация функций в implementation
Procedure OnPacket;

implementation

procedure TMain.ButtonClick(Sender: TObject); //описание обработчика для кнопки.
begin
Start := Not (Start);
Timer.Enabled := Start;
If Start
   then Button.Caption:='Стоп'
   else Button.Caption:='Старт';
end;

//процедура смена титула
procedure ChangeTitle(CharName,Title:String);
begin buf:=#$55;
      WriteS(CharName);
      WriteS(Title);
      SendToServerex(CharName);
end;



procedure TMain.TimerTimer(Sender: TObject); //обработчик таймера
var i:integer;
    Title:string;                            //Титул текущий
begin Title:='';
      for i:=1 to 16 do                          //читаем 16 символов = макс длина титула
      if ((i+Pos_i)mod(Length(Full_Title)+1))=0  //если читаем нулевой символ строки Full_Title
         then Title:=Title+' '                   //вставляем пробел   
         else Title:=Title+Full_Title[(i+Pos_i)mod(Length(Full_Title)+1)];//если нет - то вставляем символ

      Pos_i:=(Pos_i+1)mod(Length(Full_Title)+1); //меняем позицию +1 и если дошли до конца строки - идем читать сначала

      Edit.Text:=Title;
      ChangeTitle(Char_Name,Title);
end;



Constructor TMain.create(Aowner:Tcomponent); //конструктор формы
begin
  inherited; //эта cтрочка вызывает Create из родителя. т.е. из tForm
  Pos_i := 1; //Назначение переменных из секции private
  Start := False;
 
  Caption := 'Анимированный Титул v 1.0'; //назначение мени создаваемому окну, и прочих параметров
  BorderStyle := bsSizeable;
  Position := poDesigned;
  Left :=725;
  Top :=0;
  Width :=300;
  Height :=150;

 

  //Создаем обьекты, обращаем внимание на то что обьектам присваевается родитель self
  //self = этот создаваемый екземпляр формы.
  //при назначении родителя, потомки "погибают" вместе с ним.


  //пример титула будет выводиться сюда
  Edit := TEdit.Create(self);
  Edit.Parent := self;
  Edit.Left := 100;
  Edit.Top := 10;
  Edit.Width := 102;

  //таймер
  Timer := TTimer.Create(self);
  Timer.Interval := 200;
  Timer.Enabled := False;
  Timer.onTimer := TimerTimer;

  Button := TButton.Create(self);
  Button.Parent := self;
  Button.Caption := 'Старт';
  Button.Left := 10;
  Button.Top := 10;
  Button.onClick := ButtonClick; 
end;

procedure Init;
Begin
                                            //напоминаю что main:tmain а не main:tform!!!
Main := tmain.create(nil); //создание екземпляра tmain. единственное место где не назначаеться родитель.
main.show;                         //отображение екземпляра tmain 
End;


procedure Free;
Begin
Main.Destroy;  //уничтожение екземпляра Tmain ВМЕСТЕ с привязаными к нему child объектами
End;


procedure OnPacket;
Begin

End;

end.
